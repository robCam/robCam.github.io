import Button from './button';
import Spinner from './spinner';
import Dialog from './dialog';
import loadArticles from './load';

/**
 * The global namespace.
 * @namespace RCAM
 */
window.RCAM = window.RCAM || {};

/**
 * Obtains the current version of the app.
 * @memberof RCAM
 * @return {String} The app version
 * @example
 * RCAM.version(); //e.g. 'v1.0.0'
 */
window.RCAM.version = () => '@VERSION@';

const wrapper = document.querySelector('.wrapper');
const wrapperInner = wrapper.querySelector('.wrapper__inner');

const fetchBttn = new Button({
    el: '.button',
    hoverStateStyle: 'js-button--hover',
    hideStateStyle: 'js-button--hide'
}, {
    onPointerEndCallback() {
        loadArticles();
        this.toggleState();
    }
});

const spinner = new Spinner({
    el: '#loading-spinner',
    showStateStyle: 'js-loading-show',
    hideStateStyle: 'js-loading-hide',
    spinStateStyle: 'js-loading',
    spinFastStateStyle: 'js-loading--fast',
    spinSlowStateStyle: 'js-loading--slow'
});

const dialog = new Dialog({
    el: '#main-dialog',
    titleEl: '#main-dialog-title',
    activeStateStyle: 'js-main-dialog--active'
}, {
    animate: true,
    transitionEndCallback() {
        dialogCancelBttn.updateBounds();
    }
});

const dialogCancelBttn = new Button({
    el: '.main-dialog__bttn--cancel',
    hoverStateStyle: 'js-main-dialog__bttn--active'
}, {
    onPointerEndCallback() {
        dialog.toggleState();
        fetchBttn.toggleState();
    }
});

const popupDialog = new Dialog({
    el: '#popup-dialog-tray',
    titleEl: '#popup-dialog-tray-title',
    activeStateStyle: 'js-popup-dialog-tray--active'
}, {
    transitionEndCallback() {
        popupDialogCancelBttn.updateBounds();
        popupDialogRetryBttn.updateBounds();
    }
});

const popupDialogCancelBttn = new Button({
    el: '.popup-dialog__bttn--cancel',
    hoverStateStyle: 'js-popup-dialog__bttn--active'
}, {
    onPointerEndCallback() {
        popupDialog.toggleState();
        wrapper.classList.add('wrapper--scrollable');
    }
});

const popupDialogRetryBttn = new Button({
    el: '.popup-dialog__bttn--retry',
    hoverStateStyle: 'js-popup-dialog__bttn--active'
}, {
    onPointerEndCallback() {
        popupDialog.toggleState();
        loadArticles();
        wrapperInner.classList.remove('wrapper__inner--active');
    }
});

export { spinner, dialog, popupDialog };
